module CompIgual4bits (
    input [3:0] A,
    input [3:0] B,
    output S
);

    wire x0, x1, x2, x3; 

    // Operação XNOR bit a bit (1 se forem iguais)
    assign x0 = ~(A[0] ^ B[0]); 
    assign x1 = ~(A[1] ^ B[1]);
    assign x2 = ~(A[2] ^ B[2]);
    assign x3 = ~(A[3] ^ B[3]);

    // Lógica AND: S só é 1 se todos os bits forem iguais
    assign S = x0 & x1 & x2 & x3; 

endmodule
