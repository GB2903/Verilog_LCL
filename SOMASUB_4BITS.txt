module SomaSub4bits (
    input [3:0] A,
    input [3:0] B,
    input Op, // Op = 0 para Soma, Op = 1 para Subtração
    output [3:0] S,
    output Cout
);

    // Fio interno para receber o operando B modificado (inversão controlada)
    wire [3:0] B_modificado; 

    // Passo 1: Se Op=1 (Subtração), inverte os bits de B usando XOR. Se Op=0, mantém B.
    assign B_modificado[0] = B[0] ^ Op;
    assign B_modificado[1] = B[1] ^ Op;
    assign B_modificado[2] = B[2] ^ Op;
    assign B_modificado[3] = B[3] ^ Op;

    // Passo 2: Instanciar o Somador de 4 bits reaproveitando o módulo do experimento anterior
    // Passamos o Op direto no Cin para fazer o incremento de +1 do Complemento de 2
    Somador4bits somador_inst (
        .A(A),
        .B(B_modificado),
        .Cin(Op),
        .S(S),
        .Cout(Cout)
    );

endmodule
